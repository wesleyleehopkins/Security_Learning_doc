Useful One-Liners

# Find all IPs in a log file
grep -oE '[0-9]{1,3}(\.[0-9]{1,3}){3}' file.log | sort | uniq -c | sort -rn

# Find all URLs in a file
grep -oP 'https?://[^\s"]+' file.log | sort -u

# Watch for failed SSH attempts live
sudo tail -f /var/log/auth.log | grep -i 'failed\|invalid'

# Count failed SSH attempts by IP
sudo grep 'Failed password' /var/log/auth.log | awk '{print $11}' | sort | uniq -c | sort -rn

# Find hidden files in directory
ls -la /var/www/html/

# Suppress permission denied errors in find
find / -type f -name '*.bak' 2>/dev/null

# Check who is currently logged in
w

# See full process tree
ps auxf

# Elevate to root for lab work
sudo su
sudo -i
